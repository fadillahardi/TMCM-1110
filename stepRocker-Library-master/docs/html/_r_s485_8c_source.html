00070 void CSP_USART1Handler(void)
00071 {
00072   U32_T IrqFlags;
00073   int i;
00074 
00075   //See what caused the interrupt. Looks a little bit strange,
00076   //but this way it works (like it is done in the Samsung example
00077   //code).
00078   IrqFlags=USART1->SR & USART1->IMSCR;
00079 
00080   //Receive interrupt
00081   if(IrqFlags & USART_RXRDY)
00082   {
00083     //Ignore RS485 echo (which occurs when RS485 is set to
00084     //send mode).
00085     if(IS_RS485_SENDING())
00086     {
00087       i=USART1->RHR;
00088     }
00089     else
00090     {
00091       //Copy received character into the receive FIFO
00092       i=UART1RxWritePtr+1;
00093       if(i==UART_BUFFER_SIZE) i=0;
00094 
00095       if(i!=UART1RxReadPtr)
00096       {
00097         UART1RxBuffer[UART1RxWritePtr]=USART1->RHR;
00098         UART1RxWritePtr=i;
00099       }
00100 
00101       //Empfangs-Timeout auf Startwert setzen
00102       UART1TimeoutTimer=UART_TIMEOUT_VALUE;
00103 
00104       //Set transmit delay time to its start value
00105       UART1TransmitDelayTimer=UART1TransmitDelay;
00106     }
00107   }
00108 
00109   //Transmitter ready interrupt
00110   if(IrqFlags & USART_TXRDY)
00111   {
00112     //Start sending when transmit delay time has elapsed
00113     if(UART1TransmitDelayTimer==0)
00114     {
00115       //Send when there is at least one character in the FIFO
00116       if(UART1TxWritePtr!=UART1TxReadPtr)
00117       {
00118         //Set RS485 to send mode
00119         SET_RS485_SEND_MODE();
00120 
00121         //put next character into the transmit register
00122         USART1->THR=UART1TxBuffer[UART1TxReadPtr++];
00123         if(UART1TxReadPtr==UART_BUFFER_SIZE) UART1TxReadPtr=0;
00124       }
00125       else
00126       {
00127         //Switch off the transmit interrupt and switch on the transmitter empty
00128         //interrupt when there are no more characters to send.
00129         //in the transmit FIFO.
00130         USART1->IMSCR=USART_RXRDY|USART_TXEMPTY;
00131       }
00132     }
00133   }
00134 
00135   //Transmitter empty interrupt
00136   if(IrqFlags & USART_TXEMPTY)
00137   {
00138     //The very last bit has been sent, so switch RS485 back to receive
00139     //mode and turn off transmitter empty interrupt.
00140     SET_RS485_RECV_MODE();
00141     USART1->IMSCR=USART_RXRDY;
00142   }
00143 }

00164   //Configure Pins P1.03 and P1.04 as USART1 pins
00165   CSP_IOFunctionConfigure(GROUP1, IO_Port4, IOCONF_F1 );
00166   CSP_IOFunctionConfigure(GROUP1, IO_Port3, IOCONF_F1 );

00168 
00169   //Select RS485 receive mode
00170   SET_RS485_RECV_MODE();
00171 
00172   //Configure USART1
00173   USART1->CEDR = USART_CLKEN;    //UART1 Clock enable
00174   USART1->SRR = USART_SWRST;     // software reset
00175   USART1->BRGR = BaudrateTable[Baudrate];  //baud rate divider
00176   USART1->MR = USART_CHRL_8|USART_PAR_NO|USART_NBSTOP_1;  //8N1
00177   USART1->CR= USART_RXEN|USART_TXEN;  //receiver enable, transmitter enable
00178   USART1->IMSCR=USART_RXRDY;  //receive interrupt enable
00179 
00180   //Configure USART1 interrupt in the NVIC
