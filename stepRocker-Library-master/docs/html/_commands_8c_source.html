00157 void ProcessCommand(void)
00158 {
00159   UCHAR Byte;
00160   UCHAR Checksum;
00161   UCHAR i;
00162 
00163   //**Send answer for the last command**
00164   if(TMCLCommandState==TCS_UART)  //via UART
00165   {
00166     if(TMCLReplyFormat==RF_STANDARD)
00167     {
00168       Checksum=ModuleConfig.SerialHostAddress+ModuleConfig.SerialModuleAddress+
00169                ActualReply.Status+ActualReply.Opcode+
00170                ActualReply.Value.Byte[3]+
00171                ActualReply.Value.Byte[2]+
00172                ActualReply.Value.Byte[1]+
00173                ActualReply.Value.Byte[0];
00174 
00175       WriteRS485(ModuleConfig.SerialHostAddress);
00176       WriteRS485(ModuleConfig.SerialModuleAddress);
00177       WriteRS485(ActualReply.Status);
00178       WriteRS485(ActualReply.Opcode);
00179       WriteRS485(ActualReply.Value.Byte[3]);
00180       WriteRS485(ActualReply.Value.Byte[2]);
00181       WriteRS485(ActualReply.Value.Byte[1]);
00182       WriteRS485(ActualReply.Value.Byte[0]);
00183       WriteRS485(Checksum);
00184     }
00185     else if(TMCLReplyFormat==RF_SPECIAL)
00186     {
00187       for(i=0; i<9; i++)
00188       {
00189         WriteRS485(SpecialReply[i]);
00190       }
00191     }
00192   }
00193   else if(TMCLCommandState==TCS_UART_ERROR)  //check sum of the last command has been wrong
00194   {
00195     ActualReply.Opcode=0;
00196     ActualReply.Status=REPLY_CHKERR;
00197     ActualReply.Value.Int32=0;
00198 
00199     Checksum=ModuleConfig.SerialHostAddress+ModuleConfig.SerialModuleAddress+
00200              ActualReply.Status+ActualReply.Opcode+
00201              ActualReply.Value.Byte[3]+
00202              ActualReply.Value.Byte[2]+
00203              ActualReply.Value.Byte[1]+
00204              ActualReply.Value.Byte[0];
00205 
00206     WriteRS485(ModuleConfig.SerialHostAddress);
00207     WriteRS485(ModuleConfig.SerialModuleAddress);
00208     WriteRS485(ActualReply.Status);
00209     WriteRS485(ActualReply.Opcode);
00210     WriteRS485(ActualReply.Value.Byte[3]);
00211     WriteRS485(ActualReply.Value.Byte[2]);
00212     WriteRS485(ActualReply.Value.Byte[1]);
00213     WriteRS485(ActualReply.Value.Byte[0]);
00214     WriteRS485(Checksum);
00215   }
00216 
00217   //Reset state (answer has been sent now)
00218   TMCLCommandState=TCS_IDLE;
00219   TMCLReplyFormat=RF_STANDARD;
00220 
00221   //Generate a system reset if requested by the host
00222   if(ResetRequested) CSP_NVICGenerateSystemReset();
00223 
00224   //**Try to get a new command**
00225   if(ReadRS485(&Byte))  //Get data from UART
00226   {
00227     if(CheckUARTTimeout()) UARTCount=0;  //discard everything when there has been a command timeout
00228     UARTCmd[UARTCount++]=Byte;
00229 
00230     if(UARTCount==9)  //Nine bytes have been received without timeout
00231     {
00232       UARTCount=0;
00233 
00234       if(UARTCmd[0]==ModuleConfig.SerialModuleAddress)  //Is this our addresss?
00235       {
00236         Checksum=0;
00237         for(i=0; i<8; i++) Checksum+=UARTCmd[i];
00238 
00239         if(Checksum==UARTCmd[8])  //Is the checksum correct?
00240         {
00241           ActualCommand.Opcode=UARTCmd[1];
00242           ActualCommand.Type=UARTCmd[2];
00243           ActualCommand.Motor=UARTCmd[3];
00244           ActualCommand.Value.Byte[3]=UARTCmd[4];
00245           ActualCommand.Value.Byte[2]=UARTCmd[5];
00246           ActualCommand.Value.Byte[1]=UARTCmd[6];
00247           ActualCommand.Value.Byte[0]=UARTCmd[7];
00248 
00249           TMCLCommandState=TCS_UART;
00250 
00251           UARTCount=0;
00252         }
00253         else TMCLCommandState=TCS_UART_ERROR;  //Checksum wrong
00254       }
00255     }
00256   }
00257 
00258   //**Execute the command**
00259   //Check if a command could be fetched and execute it.
00260   if(TMCLCommandState!=TCS_IDLE && TMCLCommandState!=TCS_UART_ERROR) ExecuteActualCommand();
00261 }
